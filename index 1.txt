  renderModal();
  document.getElementById('productModal').classList.add('open');
}

function renderModal(){
  const p=selectedProduct;
  const initVal=initialPayment||0;
  document.getElementById('modalContent').innerHTML=`
    <div class="modal-handle"></div>
    <button class="close-modal" onclick="closeModal()">✕</button>
    <div class="modal-product-emoji">${p.emoji}</div>
    <div style="font-size:12px;color:var(--ez-muted);text-transform:uppercase;letter-spacing:.5px">${p.category}</div>
    <div class="modal-title">${p.name}</div>
    <div class="modal-price">${fmt(p.price)}</div>
    <p class="modal-desc">${p.desc}</p>
    <div class="simulator">
      <div class="sim-title">⚡ Elige tu plan de pago</div>
      <div class="initial-row">
        <div class="initial-label">Cuota inicial:</div>
        <input class="initial-input" type="number" placeholder="0" value="${initVal}" oninput="setInitial(this.value)" min="0" max="${p.price}" step="5000">
        <div class="initial-label" style="font-weight:600;color:var(--ez-blue)">${fmt(initVal)}</div>
      </div>
      <div class="sim-grid">
        ${PLANES.map((pl,i)=>`
          <div class="sim-option${i===selectedPlan?' active':''}" onclick="setPlan(${i})">
            <div class="label">${pl.label}</div>
            <div class="amount">${fmt(cuotaAmount(p.price,i,initVal))}</div>
            <div class="weeks">${pl.cuotas===1?'Pago único':pl.cuotas+' cuotas'}</div>
          </div>
        `).join('')}
      </div>
    </div>
    <button class="add-cart-btn" onclick="addToCart()">AGREGAR AL PEDIDO <span>→</span></button>
  `;
}

function setPlan(i){selectedPlan=i;renderModal();}
function setInitial(v){
  initialPayment=Math.min(Math.max(0,parseInt(v)||0),selectedProduct.price);
  renderModal();
}

function closeModal(e){
  if(!e||e.target===document.getElementById('productModal')){
    document.getElementById('productModal').classList.remove('open');
  }
}

function addToCart(){
  const p=selectedProduct;
  const plan=PLANES[selectedPlan];
  cart.push({
    id:Date.now(),product:p,
    planLabel:plan.label,cuotas:plan.cuotas,
    cuotaAmt:cuotaAmount(p.price,selectedPlan,initialPayment),
    initial:initialPayment,total:p.price
  });
  document.getElementById('cartCount').textContent=cart.length;
  document.getElementById('productModal').classList.remove('open');
  showToast('✓ Agregado al pedido');
}

function renderCart(){
  const el=document.getElementById('cartItems');
  const footer=document.getElementById('cartFooter');
  if(cart.length===0){
    el.innerHTML='<div class="cart-empty">Tu carrito está vacío.<br><br>Explora el catálogo y elige tus productos 🛍️</div>';
    footer.innerHTML='';return;
  }
  el.innerHTML=cart.map(item=>`
    <div class="cart-item">
      <div class="cart-emoji">${item.product.emoji}</div>
      <div class="cart-item-info">
        <div class="cart-item-name">${item.product.name}</div>
        <div class="cart-item-plan">Plan ${item.planLabel} · ${item.cuotas===1?'Contado':item.cuotas+' cuotas de '+fmt(item.cuotaAmt)}</div>
        ${item.initial>0?`<div style="font-size:11px;color:var(--ez-muted)">Inicial: ${fmt(item.initial)}</div>`:''}
      </div>
      <div>
        <div class="cart-item-price">${fmt(item.total)}</div>
        <button class="cart-remove" onclick="removeFromCart(${item.id})">✕</button>
      </div>
    </div>
  `).join('');
  const total=cart.reduce((s,i)=>s+i.total,0);
  footer.innerHTML=`
    <div class="cart-total">
      <div><div class="cart-total-label">Total pedido</div><div style="font-size:12px;color:#9CA3AF">${cart.length} producto${cart.length>1?'s':''}</div></div>
      <div class="cart-total-amount">${fmt(total)}</div>
    </div>
    <button class="checkout-btn" onclick="sendWhatsApp()">ENVIAR PEDIDO POR WHATSAPP ↗</button>
    <div class="wa-note"><div class="wa-dot"></div>Se abrirá WhatsApp Business de Electro Z</div>
  `;
}

function removeFromCart(id){
  cart=cart.filter(i=>i.id!==id);
  document.getElementById('cartCount').textContent=cart.length;
  renderCart();
}

function sendWhatsApp(){
  const name=document.getElementById('clientName').value.trim();
  const phone=document.getElementById('clientPhone').value.trim();
  const city=document.getElementById('clientCity').value.trim();
  let msg='🛒 *Pedido Electro Z*\n\n';
  if(name) msg+=`👤 Cliente: ${name}\n`;
  if(phone) msg+=`📱 Contacto: ${phone}\n`;
  if(city) msg+=`📍 Ubicación: ${city}\n`;
  if(name||phone||city) msg+='\n';
  cart.forEach(item=>{
    msg+=`• ${item.product.emoji} *${item.product.name}* — ${fmt(item.total)}\n`;
    msg+=`  Plan: ${item.planLabel}, ${item.cuotas===1?'Contado':item.cuotas+' cuotas de '+fmt(item.cuotaAmt)}\n`;
    if(item.initial>0) msg+=`  Inicial: ${fmt(item.initial)}\n`;
    msg+='\n';
  });
  const total=cart.reduce((s,i)=>s+i.total,0);
  msg+=`💰 *Total: ${fmt(total)}*\n\nQuedo pendiente de su confirmación. ¡Gracias! 🙏`;
  
  const newOrder={
    id:Date.now(),
    client:name||'Cliente web',
    phone:phone||'No indicado',
    items:cart.map(i=>i.product.name),
    plan:cart.map(i=>`${i.planLabel} ${i.cuotas} cuotas`).join(', '),
    total:total,
    status:'pending',
    date:new Date().toLocaleDateString('es-CO',{day:'numeric',month:'short'})
  };
  orders.unshift(newOrder);
  saveOrders();
  
  window.open('https://wa.me/'+WA_NUMBER+'?text='+encodeURIComponent(msg),'_blank');
  showToast('✓ Pedido enviado por WhatsApp');
}

let formVisible=false;
function toggleForm(){
  formVisible=!formVisible;
  document.getElementById('addForm').style.display=formVisible?'block':'none';
}

function addProduct(){
  const name=document.getElementById('f-name').value.trim();
  const price=parseInt(document.getElementById('f-price').value)||0;
  const cat=document.getElementById('f-cat').value;
  const emoji=document.getElementById('f-emoji').value||'📦';
  const desc=document.getElementById('f-desc').value.trim();
  if(!name||!price){showToast('Completa nombre y precio');return;}
  products.push({id:Date.now(),name,category:cat,price,emoji,desc:desc||'Producto de Electro Z'});
  saveProducts();
  document.getElementById('f-name').value='';
  document.getElementById('f-price').value='';
  document.getElementById('f-emoji').value='';
  document.getElementById('f-desc').value='';
  toggleForm();
  renderAdmin();renderFilters();renderProducts();
  showToast('✓ Producto agregado');
}

function deleteProduct(id){
  if(!confirm('¿Eliminar este producto?')) return;
  products=products.filter(p=>p.id!==id);
  saveProducts();
  renderAdmin();renderFilters();renderProducts();
  showToast('Producto eliminado');
}

function showToast(msg){
  const t=document.getElementById('toast');
  t.textContent=msg;t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'),2800);
}
